class CustomWorld {
  _browser = scope.browser
  _page = null

  async getPage () {
    if (!this._page) {
      this._page = await this._browser.newPage();
    }
    return this._page
  }

  async closePage () {
    if (this._page) {
      await this._page.close()
      this._page = null
    }
  }
}
